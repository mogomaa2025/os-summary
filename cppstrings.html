شرح دوال التعامل مع النصوص

myToUpper

char myToUpper(char c) {
    if (c >= 'a' && c <= 'z') {
        return c - ('a' - 'A');
    }
    return c;
}
  • char myToUpper(char c): تعريف الدالة، تأخذ حرفًا وتعيد حرفًا آخر.
  • if (c >= 'a' && c <= 'z'): الشرط للتحقق مما إذا كان الحرف صغيرًا (small letter).
  • return c - ('a' - 'A');: يتم طرح الفرق بين قيم ASCII للأحرف الصغيرة والكبيرة (32) لتحويل الحرف إلى كبير.
  • return c;: إذا لم يكن الحرف صغيرًا، يتم إرجاعه كما هو.

myToLower

char myToLower(char c) {
    if (c >= 'A' && c <= 'Z') {
        return c + ('a' - 'A');
    }
    return c;
}
  • if (c >= 'A' && c <= 'Z'): الشرط للتحقق مما إذا كان الحرف كبيرًا (capital letter).
  • return c + ('a' - 'A');: يتم إضافة الفرق (32) لتحويل الحرف إلى صغير.
  • return c;: إذا لم يكن الحرف كبيرًا، يتم إرجاعه كما هو.

myStrcat

char* myStrcat(char* destination, char* source) {
    int destLen = strlen(destination);
    int i = 0;
    while (source[i] != '\0') {
        destination[destLen + i] = source[i];
        ++i;
    }
    destination[destLen + i] = '\0';
    return destination;
}
  • int destLen = strlen(destination);: حساب طول السلسلة الأصلية (destination) لتحديد مكان بدء الإضافة.
  • while (source[i] != '\0'): المرور على كل حرف في السلسلة المصدر (source) حتى الوصول للنهاية.
  • destination[destLen + i] = source[i];: نسخ الحرف الحالي من `source` إلى نهاية `destination`.
  • destination[destLen + i] = '\0';: بعد انتهاء النسخ، يتم إضافة الحرف `\0` لإنهاء السلسلة الجديدة المدمجة.
  • return destination;: إرجاع المؤشر إلى السلسلة المدمجة.

myStrcpy

char* myStrcpy(char* destination, char* source) {
    char* ptr = destination;
    while (*source != '\0') {
        *ptr++ = *source++;
    }
    *ptr = '\0';
    return destination;
}
  • char* ptr = destination;: استخدام مؤشر إضافي للحفاظ على بداية `destination`.
  • *ptr++ = *source++;: في خطوة واحدة، يتم نسخ الحرف الذي يشير إليه `source` إلى المكان الذي يشير إليه `ptr`، ثم يتم تحريك كلا المؤشرين إلى الحرف التالي.
  • *ptr = '\0';: بعد انتهاء الحلقة، يتم إضافة حرف النهاية `\0`.
  • return destination;: إرجاع المؤشر الأصلي للسلسلة المنسوخة.

myStrcmp

int myStrcmp(char* str1, char* str2) {
    while (*str1 && (*str1 == *str2)) {
        str1++;
        str2++;
    }
    return *(unsigned char*)str1 - *(unsigned char*)str2;
}
  • while (*str1 && (*str1 == *str2)): استمر في الحلقة طالما أن الحرف الحالي في السلسلة الأولى ليس النهاية (`\0`) ومتساوٍ مع الحرف المقابل في السلسلة الثانية.
  • str1++; str2++;: إذا كانت الأحرف متطابقة، انتقل إلى الحرف التالي في كلتا السلسلتين.
  • return *(unsigned char*)str1 - *(unsigned char*)str2;: عند العثور على أول اختلاف، أرجع الفرق بين قيم ASCII للحرفين.
    • إذا كانت النتيجة 0، فالسلسلتان متطابقتان.
    • إذا كانت موجبة، فإن `str1` أكبر.
    • إذا كانت سالبة، فإن `str2` أكبر.

طريقة بديلة (ترجع -1 أو 0 أو 1)

int myStrcmp_alt(const char* str1, const char* str2) {
    while (*str1 && *str2) {
        if (*str1 != *str2) {
            return ((*(unsigned char*)str1 < *(unsigned char*)str2) ? -1 : 1);
        }
        str1++;
        str2++;
    }
    if (*str1 == *str2) { // Both are '\0'
        return 0;
    } else if (*str1 == '\0') { // str1 is shorter
        return -1;
    } else { // str2 is shorter
        return 1;
    }
}

الشرح:

  • الهدف هو نفسه، ولكن الإرجاع يكون فقط -1، 0، أو 1 لتحديد العلاقة (أصغر، يساوي، أكبر).
  • while (*str1 && *str2): أكمل المقارنة طالما لم تصل أي من السلسلتين إلى نهايتها.
  • بعد الحلقة، يتم التحقق من الحالات النهائية (هل انتهت السلسلتان معًا أم أن إحداهما أقصر).

myIsAlpha

int myIsAlpha(char c) {
    return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z');
}
  • تتحقق مما إذا كان الحرف المدخل يقع ضمن نطاق الحروف الأبجدية الكبيرة أو الصغيرة.
  • ترجع 1 (true) إذا كان حرفًا، و 0 (false) إذا لم يكن كذلك.

myIsDigit

int myIsDigit(char c) {
    return (c >= '0' && c <= '9');
}
  • تتحقق مما إذا كان الحرف المدخل يقع ضمن نطاق الأرقام من '0' إلى '9'.
  • ترجع 1 (true) إذا كان رقمًا، و 0 (false) إذا لم يكن كذلك.

myIsAlnum

int myIsAlnum(char c) {
    return myIsAlpha(c) || myIsDigit(c);
}
  • تجمع بين الدالتين السابقتين للتحقق مما إذا كان الحرف حرفًا أبجديًا أو رقمًا.
  • ترجع 1 (true) إذا كان الحرف أو الرقم، و 0 (false) خلاف ذلك.

myIsPrint

int myIsPrint(char c) {
    return (c >= 32 && c <= 126);
}
  • تتحقق مما إذا كانت قيمة ASCII للحرف تقع في نطاق الأحرف القابلة للطباعة.
  • النطاق يبدأ من 32 (المسافة) وينتهي عند 126 (~).
  • ترجع 1 (true) إذا كان الحرف قابلاً للطباعة، و 0 (false) إذا لم يكن كذلك.
